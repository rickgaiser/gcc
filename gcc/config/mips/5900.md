;; R5900 instruction patterns and pipeline tuning.

(define_insn_reservation "r5900_hilo1" 1
  (eq_attr "type" "mfhi1,mflo1,mthi1,mtlo1")
  "imuldiv*3")

;; RSQRT
(define_insn "rsqrtsf"
  [(set (match_operand:SF 0 "register_operand" "=f")
	(div:SF (match_operand:SF 1 "register_operand" "f")
		  (sqrt:SF (match_operand:SF 2 "register_operand" "f"))))]
  "TARGET_MIPS5900"
  "rsqrt.s\t%0,%1,%2"
  [(set_attr "type" "frsqrt")
   (set_attr "mode" "SF")])

;; MIN and MAX
(define_insn "sminsf3"
  [(set (match_operand:SF 0 "register_operand" "=f")
	(smin:SF (match_operand:SF 1 "register_operand" "f")
		 (match_operand:SF 2 "register_operand" "f")))]
  "TARGET_MIPS5900"
  "min.s\t%0,%1,%2"
  [(set_attr "type" "fadd")
   (set_attr "mode" "SF")])

(define_insn "smaxsf3"
  [(set (match_operand:SF 0 "register_operand" "=f")
	(smax:SF (match_operand:SF 1 "register_operand" "f")
		 (match_operand:SF 2 "register_operand" "f")))]
  "TARGET_MIPS5900"
  "max.s\t%0,%1,%2"
  [(set_attr "type" "fadd")
   (set_attr "mode" "SF")])
